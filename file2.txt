kind: Service
metadata:
  name: drink-service
  namespace: foo

  selector:
    app: drink-app
    version: blue   # switch to green during rollout
  ports:
  - port: 80
    targetPort: 8080


stage("Blue-Green Switch") {
    steps {
        script {
            def active = sh(script: "kubectl get svc drink-service -n ${NAMESPACE} -o jsonpath='{.spec.selector.version}'", returnStdout: true).trim()
            def target = (active == "blue") ? "green" : "blue"

            echo "Active color: ${active}, deploying to: ${target}"

            sh """
                sed -i 's|replace|${FINALIAMGE}|g' ${target}.yaml
                kubectl apply -f ${target}.yaml -n ${NAMESPACE}
            """

            // Switch service to new version
            sh """
                kubectl patch svc drink-service -n ${NAMESPACE} -p '{"spec":{"selector":{"app":"drink-app","version":"${target}"}}}'
            """
        }
    }
}


pipeline {
    agent any
    stages {
        stage('Vault Secrets') {
            steps {
                script {
                    def secrets = [
                        [
                            path: 'secret/jenkins/k8s',

                            secretValues: [
                                [envVar: 'CLUSTERNAME', vaultKey: 'CLUSTERNAME'],
                                [envVar: 'NAMESPACE', vaultKey: 'NAMESPACE'],
                                [envVar: 'FINALIAMGE', vaultKey: 'FINALIAMGE']
                            ]
                        ]
                    ]

                    def config = [
                        vaultUrl: 'http://my-vault.com:8200',
                        vaultCredentialId: 'vault-approle'
                    ]

                    withVault([configuration: config, vaultSecrets: secrets]) {
                        // Now CLUSTERNAME, NAMESPACE, FINALIAMGE are available as env vars
                        echo "Cluster: $CLUSTERNAME"
                        echo "Namespace: $NAMESPACE"
                        echo "Image: $FINALIAMGE"
                    }
                }
            }
        }

        stage('Deploy K8s') {

                    withKubeConfig(
                        caCertificate: '',
                        clusterName: "${CLUSTERNAME}",

                    ) {

                        """
                    }
                }
            }
        }
    }
